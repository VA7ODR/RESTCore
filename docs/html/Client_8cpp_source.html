12RESTCore::Client::ParsedUrl RESTCore::Client::parseUrl(const std::string& url) {
13 // Very small/basic URL parser for http/https
14 // Supports: http(s)://host[:port][/path?query]
15 static const std::regex re(R"(^([Hh][Tt][Tt][Pp][Ss]?)://([^/:]+)(?::(\d+))?(\/.*)?$")");
16 std::smatch m;
17 if (!std::regex_match(url, m, re)) {
18 throw std::invalid_argument("Unsupported or invalid URL: " + url);
19 }
20 ParsedUrl p;
21 const auto scheme = m[1].str();
22 p.https = (scheme == "https" || scheme == "HTTPS" || scheme == "Https");
23 p.host = m[2].str();
24 p.port = m[3].matched ? m[3].str() : (p.https ? "443" : "80");
25 p.target = m[4].matched ? m[4].str() : std::string("/");
26 return p;
27}

108std::tuple<unsigned, RESTCore::Client::Response>
109RESTCore::Client::request(bool https,
110 http::verb method,
111 const std::string& host,
112 const std::string& port,
113 const std::string& target,
114 const Headers& headers,
115 const std::string* body,
116 const std::string* content_type) {
117 net::io_context ioc;
118
119 // Resolve
120 tcp::resolver resolver{ioc};
121 auto const results = resolver.resolve(host, port);
122
123 // Common HTTP request setup
124 http::request<http::string_body> req{method, target, 11};
125 req.set(http::field::host, host);
126 req.set(http::field::user_agent, std::string{"HTTPClient/1.0 (Boost.Beast)"});
127 for (const auto& kv : headers) {
128 req.set(kv.first, kv.second);
129 }
130 if (body) {
131 req.set(http::field::content_type, content_type ? *content_type : std::string{"application/octet-stream"});
132 req.body() = *body;
133 req.prepare_payload(); // sets Content-Length
134 }
135
136 Response res;
137
138 if (!https) {
139 // Plain HTTP
140 beast::tcp_stream stream{ioc};
141 stream.connect(results);
142
143 http::write(stream, req);
144 beast::flat_buffer buffer;
145 http::read(stream, buffer, res);
146
147 beast::error_code ec;
148 stream.socket().shutdown(tcp::socket::shutdown_both, ec);
149 // ignore shutdown errors
150
151 } else {
152 // HTTPS (TLS)
153 ssl::context ctx{ssl::context::tls_client};
154 ctx.set_default_verify_paths();
155 ctx.set_verify_mode(ssl::verify_peer);
156
157 beast::ssl_stream<beast::tcp_stream> stream{ioc, ctx};
158
159 // SNI
160 if(! SSL_set_tlsext_host_name(stream.native_handle(), host.c_str())) {
161 beast::error_code ec{static_cast<int>(::ERR_get_error()), net::error::get_ssl_category()};
162 throw beast::system_error{ec};
163 }
164
165 // Connect
166 beast::get_lowest_layer(stream).connect(results);
167
168 // TLS handshake
169 stream.handshake(ssl::stream_base::client);
170
171 http::write(stream, req);
172 beast::flat_buffer buffer;
173 http::read(stream, buffer, res);
174
175 beast::error_code ec;
176 // Graceful TLS shutdown (ignore errors to avoid nasty EOFs from servers)
177 stream.shutdown(ec);
178 }
179
180 return {res.result_int(), std::move(res)};
181}

28void RESTCore::Server::start() {
29 // Start HTTP listeners
30 for (const auto& cfg : http_cfgs_) {
31 auto rt = std::make_unique<ListenerRuntime>();
32 rt->running.store(true);
33 rt->address = cfg.address;
34 rt->port = cfg.port;
35 auto cb = callback_;
36 rt->thread = std::thread([rt_ptr = rt.get(), cb, cfg]() mutable {
37 http_accept_loop(rt_ptr, cb, cfg.address, cfg.port);
38 });
39 http_runtimes_.push_back(std::move(rt));
40 }
41
42 // Start HTTPS listeners
43 for (const auto& cfg : https_cfgs_) {
44 auto rt = std::make_unique<ListenerRuntime>();

46 rt->address = cfg.address;

56void RESTCore::Server::stop() {
57 auto wake = [](const std::string& addr, unsigned short port) {
58 try {
59 std::string connect_addr = addr;
60 if (addr == "0.0.0.0") connect_addr = "127.0.0.1";
61 else if (addr == "::") connect_addr = "::1";
62 net::io_context ioc;
63 tcp::socket sock{ioc};
64 boost::system::error_code ec;
65 auto ip = net::ip::make_address(connect_addr, ec);
66 if (ec) return; // invalid addr, give up
67 tcp::endpoint ep{ip, port};
68 sock.connect(ep, ec);
69 // immediately close; connection existing is enough to unblock accept
70 } catch (...) {
71 // ignore
72 }
73 };
74
75 for (auto& rt : http_runtimes_) {
76 bool was_running = rt->running.exchange(false);
77 if (was_running) {
78 wake(rt->address, rt->port);
79 }
80 }
81 for (auto& rt : https_runtimes_) {
82 bool was_running = rt->running.exchange(false);
83 if (was_running) {
84 wake(rt->address, rt->port);
85 }
86 }
87 for (auto& rt : http_runtimes_) {
88 if (rt->thread.joinable()) rt->thread.join();
89 }
90 for (auto& rt : https_runtimes_) {
91 if (rt->thread.joinable()) rt->thread.join();
92 }
93 http_runtimes_.clear();
94 https_runtimes_.clear();
95}

97void RESTCore::Server::http_accept_loop(ListenerRuntime* rt,
98 Callback cb,
99 std::string address,
100 unsigned short port) {
101 try {
102 net::io_context ioc;
103 tcp::endpoint ep{net::ip::make_address(address), port};
104 tcp::acceptor acceptor{ioc};
105 acceptor.open(ep.protocol());
106 acceptor.set_option(net::socket_base::reuse_address(true));
107 acceptor.bind(ep);
108 acceptor.listen(net::socket_base::max_listen_connections);
109
110 while (rt->running.load()) {
111 tcp::socket socket{ioc};
112 boost::system::error_code ec;
113 acceptor.accept(socket, ec);
114 if (ec) {
115 if (!rt->running.load()) break;
116 continue; // transient error, continue
117 }
118 // If we've been asked to stop, do not spawn a session for this wake-up connection
119 if (!rt->running.load()) {
120 boost::system::error_code ignore_ec;
121 socket.shutdown(tcp::socket::shutdown_both, ignore_ec);
122 socket.close(ignore_ec);
123 break;
124 }
125 std::thread(&RESTCore::Server::handle_http_session, std::move(socket), cb).detach();
126 }
127 } catch (const std::exception& e) {
128 // Log and exit loop
129 (void)e;
130 }
131}

133void RESTCore::Server::https_accept_loop(ListenerRuntime* rt,
134 Callback cb,
135 std::string address,
136 unsigned short port,
137 std::string cert_file,
138 std::string key_file) {
139 try {
140 net::io_context ioc;
141 tcp::endpoint ep{net::ip::make_address(address), port};
142 tcp::acceptor acceptor{ioc};
143 acceptor.open(ep.protocol());
144 acceptor.set_option(net::socket_base::reuse_address(true));
145 acceptor.bind(ep);
146 acceptor.listen(net::socket_base::max_listen_connections);
147
148 auto ctx = std::make_shared<ssl::context>(ssl::context::tls_server);
149 ctx->set_options(ssl::context::default_workarounds | ssl::context::no_sslv2 | ssl::context::no_sslv3);
150 ctx->use_certificate_chain_file(cert_file);
151 ctx->use_private_key_file(key_file, ssl::context::file_format::pem);
152
153 while (rt->running.load()) {
154 tcp::socket socket{ioc};
155 boost::system::error_code ec;
156 acceptor.accept(socket, ec);
157 if (ec) {
158 if (!rt->running.load()) break;
159 continue;
160 }
161 // If stopping, avoid spawning a session for the wake-up connection
162 if (!rt->running.load()) {
163 boost::system::error_code ignore_ec;
164 socket.shutdown(tcp::socket::shutdown_both, ignore_ec);
165 socket.close(ignore_ec);
166 break;
167 }
168 std::thread(&RESTCore::Server::handle_https_session, std::move(socket), cb, ctx).detach();
169 }
170 } catch (const std::exception& e) {
171 // Log and exit loop
172 (void)e;
173 }

176static std::string remote_addr_string(const tcp::socket& s) {
177 boost::system::error_code ec;
178 auto ep = s.remote_endpoint(ec);

183void RESTCore::Server::handle_http_session(tcp::socket socket, Callback cb) {
184 try {
185 beast::tcp_stream stream{std::move(socket)};
186 beast::flat_buffer buffer;
187 RESTCore::Server::Request req;
188 http::read(stream, buffer, req);
189
190 RESTCore::Server::Response res{http::status::ok, req.version()};
191 res.set(http::field::server, std::string{"HTTPServerHost/1.0"});
192 res.keep_alive(false);
193
194 const std::string client = remote_addr_string(stream.socket());
195
196 if (cb) cb(req, res, client);
197
198 // Ensure we have content-length if body set
199 if (res.body().size() > 0 && res.find(http::field::content_length) == res.end()) {
200 res.prepare_payload();
201 }
202
203 http::write(stream, res);
204
205 boost::system::error_code ec;
206 stream.socket().shutdown(tcp::socket::shutdown_both, ec);
207 } catch (...) {
208 // swallow
209 }
210}
